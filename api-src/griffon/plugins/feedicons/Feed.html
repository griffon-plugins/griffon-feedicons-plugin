27     ACTIVITY_WINDOW("activity_window"),
28     CHECK_ALL("check_all"),
29     COMMENT_RSS("comment_rss"),
30     COMMENT_RSS_ADD("comment_rss_add"),
31     COMMENT_RSS_DELETE("comment_rss_delete"),
32     COMMENT_RSS_INFO("comment_rss_info"),
33     FEED_SOURECe("feed_source"),
34     FEED("feed"),
35     FOLDER("folder"),
36     FOLDER_ADD("folder_add"),
37     FOLDER_REMOVE("folder_remove"),
38     FOLDER_RSS("folder_rss"),
39     INFO("info"),
40     MARK_AS_READ("mark_as_read"),
41     NEXT_RSS("next_rss"),
42     NEXT_UNREAD("next_unread"),
43     POST_DELICIOUS("post_delicious"),
44     POST_TO_BLOG("post_to_blog"),
45     PREVIOUS_RSS("previous_rss"),
46     PREVIOUS_UNREAD("previous_unread"),
47     RELOAD("reload"),
48     RSS("rss"),
49     RSS_FILE("rss_file"),
50     RSS_FILE_ADD("rss_file_add"),
51     RSS_FILE_INFO("rss_file_info"),
52     RSS_FILE_REMOVE("rss_file_remove"),
53     RSS_PILL_BLUE("rss_pill_blue"),
54     RSS_PILL_ORANGE("rss_pill_orange"),
55     RSS_SITE("site"),
56     RSS_SITE_RSS("site_rss"),
57     STOP("stop"),
58     SUBSCRIBE("subscribe"),
59     UNSUBSCRIBE("unsubscribe");
60 
61     private final String description;
62 
63     Feed(@Nonnull String description) {
64         this.description = description;
65     }
66 
67     @Nonnull
68     public String getDescription() {
69         return description;
70     }
71 
72     @Nonnull
73     public String asResource(int size) {
74         requireState(size == 16 || size == 32, "Argument 'size' must be either 16 or 32.");
75         return "com/zeusboxstudio/icons/" + size + "x" + size + "/" + description + ".png";
76     }
77 
78     @Nonnull
79     public static Feed findByDescription(@Nonnull String description) {
80         if (isBlank(description)) {
81             throw new IllegalArgumentException("Description " + description + " is not a valid Feed icon description");
82         }
83 
84         for (Feed feed : values()) {
85             if (feed.description.equalsIgnoreCase(description)) {
86                 return feed;
87             }
88         }
89         throw new IllegalArgumentException("Description " + description + " is not a a valid Feed icon description");
90     }
91 }
